private final static DirectionProperty FACING = DirectionProperty.of("facing", Direction.NORTH, Direction.SOUTH, Direction.EAST, Direction.WEST);
private final static EnumProperty NORTH = EnumProperty.of("north", EnergyStorageModuleSideOptions.class);
private final static EnumProperty SOUTH = EnumProperty.of("south", EnergyStorageModuleSideOptions.class);
private final static EnumProperty EAST = EnumProperty.of("east", EnergyStorageModuleSideOptions.class);
private final static EnumProperty WEST = EnumProperty.of("west", EnergyStorageModuleSideOptions.class);
private final static EnumProperty UP = EnumProperty.of("up", EnergyStorageModuleSideOptions.class);
private final static EnumProperty DOWN = EnumProperty.of("down", EnergyStorageModuleSideOptions.class);

@Override
public void appendProperties(StateFactory.Builder<Block, BlockState> stateBuilder) {
    super.appendProperties(stateBuilder);
    stateBuilder.add(FACING);
    stateBuilder.add(NORTH);
    stateBuilder.add(SOUTH);
    stateBuilder.add(EAST);
    stateBuilder.add(WEST);
    stateBuilder.add(UP);
    stateBuilder.add(DOWN);
}

@Override
public BlockState getPlacementState(ItemPlacementContext context) {
    return this.getDefaultState().with(FACING, context.getPlayerFacing().getOpposite())
        .with(NORTH, EnergyStorageModuleSideOptions.DEFAULT)
        .with(SOUTH, EnergyStorageModuleSideOptions.DEFAULT)
        .with(EAST, EnergyStorageModuleSideOptions.DEFAULT)
        .with(WEST, EnergyStorageModuleSideOptions.DEFAULT)
        .with(UP, EnergyStorageModuleSideOptions.DEFAULT)
        .with(DOWN, EnergyStorageModuleSideOptions.DEFAULT)
}


public enum EnergyStorageModuleSideOptions implements StringIdentifiable {
    POWERIN("powerin"),
    POWEROUT("powerout"),
    DEFAULT("default"),
    ;

    String name;
    EnergyStorageModuleSideOptions(String name) {
        this.name = name;
    }
    @Override
    public String asString() {
        return this.name;
    }
}
